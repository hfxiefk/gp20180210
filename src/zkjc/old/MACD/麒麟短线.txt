{麒麟短线：论坛08年的公式}
AM:=VOL*(H+2*(O+C)+L)/6;
生命线:SUM(AM,88)/SUM(VOL,88),COLORGRAY,LINETHICK2;
年线:=EMA(C,250);
趋势线:年线-(EMA(C,3)-年线),COLORGREEN,LINETHICK2;
FL:MA(CLOSE,17)+ABS(MA(CLOSE,17)-REF(MA(CLOSE,17),1)),COLOR789456;
FS:MA(CLOSE,17)+MA(CLOSE,17)-REF(MA(CLOSE,17),1),COLOR789456;
CG:IF(MA(CLOSE,17)<FS,FS,MA(CLOSE,17)),COLOR789456;
持股线:IF(CG=FS,CG,DRAWNULL), COLORYELLOW,LINETHICK1; 
波段线:CG-(EMA(C,3)-CG),COLORMAGENTA;
